FONCTION rechercheSequentielle(tableau, element)
 compteur <- 0
   POUR i de 0 à longueur(tableau) – 1
     SI tableau[i] = element
     compteur <- compteur + 1
      RETOURNER "Il y a" + compteur + "fois" + i 
     FIN SI
   FIN POUR
  RETOURNER -1
FIN FONCTION

tableauNombres[15, 12, 36, 45, 45, 72]

RechercheNombre <- rechercheSequentielle(tableauNombres, 45)
   AFFICHER RechercheNombre   // Il y a 2 fois 45
