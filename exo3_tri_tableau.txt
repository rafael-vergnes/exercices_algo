  indexMin <- i
   POUR j de i+1 à n-1
     SI tableau[indexMin] > tableau[j]
       indexMin <- j
     FIN SI
   FIN POUR
     SI indexMin != i
       temporaire <- tableau[j]
       tableau[j] <- tableau[indexMin]
       tableau[indexMin] <- temporaire
     FIN SI
FIN POUR
